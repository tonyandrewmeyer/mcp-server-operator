   244→def _build_tool_handler(
   245→    handler: dict[str, Any],
   246→    input_schema: dict[str, Any],
   247→    properties: dict[str, Any],
   248→    required: set[str],
   249→    command_allowlist: list[str] | None = None,
   250→    tool_name: str = "",
   251→) -> Any:
   252→    """Build a tool handler function with an explicit signature matching the input schema.
   253→
   254→    FastMCP introspects the handler's signature to build the input schema, so
   255→    we need named parameters rather than **kwargs.
   256→    """
   257→
   258→    async def tool_handler(**kwargs: Any) -> list[TextContent]:
   259→        # Validate arguments against declared schema before execution.
   260→        errors = validate_arguments(kwargs, input_schema)
   261→        if errors:
   262→            TOOL_CALLS.labels(tool_name=tool_name, status="error").inc()
   263→            return [TextContent(type="text", text=f"Validation error: {'; '.join(errors)}")]
   264→
   265→        handler_type = handler["type"]
   266→        if handler_type == "exec":
   267→            # Check command allowlist if configured.
   268→            if command_allowlist is not None:
   269→                executable = handler["command"][0]
   270→                if executable not in command_allowlist:
   271→                    TOOL_CALLS.labels(tool_name=tool_name, status="error").inc()
   272→                    return [
   273→                        TextContent(
   274→                            type="text",
   275→                            text=f"Command '{executable}' is not in the allowlist",
   276→                        )
   277→                    ]
   278→            output = execute_exec_handler(handler, kwargs)
   279→        elif handler_type == "http":
   280→            output = await execute_http_handler(handler, kwargs)
   281→        else:
   282→            output = f"Unknown handler type: {handler_type}"
   283→        TOOL_CALLS.labels(tool_name=tool_name, status="success").inc()
   284→        return [TextContent(type="text", text=output)]
   285→
   286→    # Build a proper signature so FastMCP sees named parameters.
   287→    params = []
   288→    for param_name in properties:
   289→        default = inspect.Parameter.empty if param_name in required else None
   290→        params.append(
   291→            inspect.Parameter(
   292→                param_name,
   293→                inspect.Parameter.KEYWORD_ONLY,
   294→                default=default,
   295→                annotation=str,
   296→            )
   297→        )
   298→    tool_handler.__signature__ = inspect.Signature(params)  # ty: ignore[unresolved-attribute]
   299→
   300→    return tool_handler
