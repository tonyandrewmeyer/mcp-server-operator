    18→@dataclasses.dataclass
    19→class _McpRelationData:
    20→    """Schema for data written to the relation app data bag."""

    22→    mcp_definitions: str = ""
    23→
    24→
    25→class McpProvider(ops.Object):
    26→    """Manages the provider side of the ``mcp`` relation.
    27→
    28→    The provider is typically a principal charm that wants to expose
    29→    tools, prompts, and resources to an MCP server subordinate.
    30→
    31→    Usage::
    32→
    33→        class MyCharm(ops.CharmBase):
    34→            def __init__(self, framework):
    35→                super().__init__(framework)
    36→                self.mcp = McpProvider(self, "mcp")
    37→                framework.observe(self.on.mcp_relation_joined, self._on_mcp_joined)
    38→
    39→            def _on_mcp_joined(self, event):
    40→                self.mcp.set_definitions(McpDefinitions(tools=[...]))
    41→    """
    42→
    43→    def __init__(self, charm: ops.CharmBase, relation_name: str = "mcp"):
    44→        super().__init__(charm, relation_name)
    45→        self._charm = charm
    46→        self._relation_name = relation_name
    47→
    48→    def set_definitions(self, definitions: McpDefinitions) -> None:
    49→        """Publish MCP definitions on all current relations.
    50→
    51→        This writes the definitions as a JSON string to the app data bag
    52→        of every ``mcp`` relation.  Must be called by the leader unit.
    53→        """
    54→        if not self._charm.unit.is_leader():
    55→            logger.debug("Not the leader; skipping set_definitions")
    56→            return
    57→
    58→        data = _McpRelationData(mcp_definitions=definitions.to_json())
    59→        for relation in self._charm.model.relations.get(self._relation_name, []):
    60→            relation.save(data, self._charm.app)
    61→            logger.info(
    62→                "Published MCP definitions on relation %d (%d tools, %d prompts, %d resources)",
    63→                relation.id,
    64→                len(definitions.tools),
    65→                len(definitions.prompts),
    66→                len(definitions.resources),
    67→            )
    68→
    69→    def set_tools(self, tools: list[Tool]) -> None:
    70→        """Publish only tools, preserving any existing prompts and resources."""
    71→        current = self._get_current_definitions()
    72→        current.tools = tools
    73→        self.set_definitions(current)
    74→
    75→    def set_prompts(self, prompts: list[Prompt]) -> None:
    76→        """Publish only prompts, preserving any existing tools and resources."""
    77→        current = self._get_current_definitions()
    78→        current.prompts = prompts
    79→        self.set_definitions(current)
    80→
    81→    def set_resources(self, resources: list[Resource]) -> None:
    82→        """Publish only resources, preserving any existing tools and prompts."""
    83→        current = self._get_current_definitions()
    84→        current.resources = resources

    37→
    38→            def _on_relation_changed(self, event):

    40→                # definitions is a merged dict with keys: tools, prompts, resources
    41→    """
    42→
    43→    def __init__(self, charm: ops.CharmBase, relation_name: str = "mcp"):
    44→        super().__init__(charm, relation_name)
    45→        self._charm = charm
    46→        self._relation_name = relation_name
    47→
    48→    def has_definitions(self) -> bool:
    49→        """Return True if any related provider has published definitions."""
    50→        for relation in self._charm.model.relations.get(self._relation_name, []):
    51→            remote_app = relation.app
    52→            if not remote_app:
    53→                continue
    54→            data = relation.load(_McpRelationData, remote_app)
    55→            if data.mcp_definitions:
    56→                return True
    57→        return False
    58→
    59→    def collect_definitions(self) -> dict[str, Any]:
    60→        """Collect and merge MCP definitions from all related providers.
    61→
    62→        Returns a dict with keys ``tools``, ``prompts``, and ``resources``,
    63→        each containing a list of definition dicts merged from every
    64→        provider relation.
    65→        """
    66→        all_tools: list[dict[str, Any]] = []
    67→        all_prompts: list[dict[str, Any]] = []
    68→        all_resources: list[dict[str, Any]] = []
    69→
    70→        for relation in self._charm.model.relations.get(self._relation_name, []):
    71→            remote_app = relation.app
    72→            if not remote_app:
    73→                continue

    76→                continue

    96→            all_resources.extend(definitions.get("resources", []))
